packages = [ "pandas", "numpy", "matplotlib", "micropip" ] import micropip await micropip.install("https://github.com/explosion/spacy-models/releases/download/pt_core_news_sm-3.7.0/pt_core_news_sm-3.7.0-py3-none-any.whl") await micropip.install("spacy") import spacy, pandas as pd, numpy as np from collections import Counter import matplotlib.pyplot as plt import io, base64 from pyscript import Element # Elementos da página status_el = Element("output-status") table_el = Element("output-table") plot_el = Element("output-plot") loader_el = Element("loader") nlp = spacy.load("pt_core_news_sm") BASE_POS = ["NOUN","VERB","ADJ","ADV","ADP","PROPN"] def calcular_icoer(frases): registros = [] for i, sent in enumerate(nlp.pipe(frases), start=1): if len(sent) < 3: continue counts = Counter([t.pos_ for t in sent]) total = sum(counts.values()) or 1 probs = np.array([counts.get(pos,0)/total for pos in BASE_POS if counts.get(pos,0)>0]) ent = -np.sum(probs * np.log2(probs)) if probs.size>0 else 0 ent_max = np.log2(len(BASE_POS)) icoer = round(max(0,1 - (ent/ent_max if ent_max>0 else 0)),4) registros.append({ "narraton": i, "frase": sent.text, "n_palavras": len(sent), "tags": dict(counts), "icoer": icoer }) if not registros: return pd.DataFrame([{"narraton":0,"frase":"NENHUMA FRASE VÁLIDA","n_palavras":0,"tags":{},"icoer":0}]) df = pd.DataFrame(registros) summary = pd.DataFrame([{ "narraton": 0, "frase": "MÉDIA GERAL", "n_palavras": round(df["n_palavras"].mean(),2), "tags": {pos: round(df["tags"].apply(lambda d: d.get(pos,0)).mean(),2) for pos in BASE_POS}, "icoer": round(df["icoer"].mean(),4) }]) return pd.concat([summary, df], ignore_index=True) async def run_analysis(): loader_el.element.style.display = "block" status_el.write("") table_el.clear() plot_el.clear() try: raw = Element("text-input").value frases = [s.strip() for s in nlp(raw).sents if s.text.strip()] if not frases: status_el.write("Nenhuma frase encontrada.") return status_el.write("Analisando...") df = calcular_icoer(frases) table_el.write(df.to_html(index=False), append=False) df_plot = df[df["narraton"]!=0] if not df_plot.empty: fig, ax = plt.subplots(figsize=(8,4)) ax.bar(df_plot["narraton"].astype(str), df_plot["icoer"]) ax.set_ylim(0,1) ax.set_xlabel("Frase (narraton)") ax.set_ylabel("ICOER") plt.tight_layout() buf = io.BytesIO() fig.savefig(buf, format="png"); buf.seek(0) img = base64.b64encode(buf.read()).decode("utf-8") plot_el.write(f'', append=False) plt.close(fig) status_el.write("Pronto!") except Exception as e: status_el.write(f"Erro: {e}") finally: loader_el.element.style.display = "none"
