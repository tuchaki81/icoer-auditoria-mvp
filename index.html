packages = ["pandas","numpy","matplotlib","spacy"] files = ["https://github.com/explosion/spacy-models/releases/download/pt_core_news_sm-3.7.0/pt_core_news_sm-3.7.0-py3-none-any.whl"] import pandas as pd import numpy as np import spacy from collections import Counter import matplotlib.pyplot as plt import io, base64 from pyscript import Element # elementos da página status_el = Element("output-status") table_el = Element("output-table") plot_el = Element("output-plot") loader_el = Element("loader") nlp = None BASE_POS = ['NOUN','VERB','ADJ','ADV','ADP','PROPN'] # carrega o modelo (uma vez só) async def setup_spacy(): global nlp if nlp is None: status_el.write("Carregando modelo…") loader_el.element.style.display = "block" try: await spacy.cli.package.install("pt_core_news_sm-3.7.0-py3-none-any.whl") nlp = spacy.load("pt_core_news_sm") status_el.write("Modelo carregado.") except Exception as e: status_el.write(f"Erro ao carregar modelo: {e}") finally: loader_el.element.style.display = "none" # função que faz todo o fluxo de análise def calcular_icoer(frases): docs = list(nlp.pipe(frases)) regs = [] for i, doc in enumerate(docs, start=1): if len(doc)<3 or not any(t.is_alpha or t.is_digit for t in doc): continue counts = Counter(t.pos_ for t in doc) total = sum(counts.values()) vetor = [counts.get(p,0)/total if total>0 else 0 for p in BASE_POS] probs = np.array([v for v in vetor if v>0]) ent = -np.sum(probs*np.log2(probs)) if probs.size else 0 ent_max= np.log2(len(BASE_POS)) if len(BASE_POS)>1 else 1 icoer = round(max(0,1-(ent/ent_max if ent_max>0 else 0)),4) regs.append({'narraton':i,'frase':doc.text,'n_palavras':len(doc),'tags':dict(counts),'icoer':icoer}) df = pd.DataFrame(regs) if df.empty: return pd.DataFrame([{'narraton':0,'frase':'Nenhuma frase válida','n_palavras':0,'tags':{},'icoer':0}]) tag_sum = {p:round(np.mean([r['tags'].get(p,0) for r in regs]),2) for p in BASE_POS} resumo = pd.DataFrame([{'narraton':0,'frase':'MÉDIA GERAL', 'n_palavras':round(df['n_palavras'].mean(),2), 'tags':tag_sum,'icoer':round(df['icoer'].mean(),4)}]) return pd.concat([resumo, df],ignore_index=True)[['narraton','frase','n_palavras','tags','icoer']] # função chamada ao clicar no botão async def run_analysis(*args, **kwargs): status_el.clear() status_el.write("Botão clicado!") # << debug if nlp is None: await setup_spacy() if nlp is None: return table_el.clear() plot_el.clear() status_el.write("Analisando…") loader_el.element.style.display = "block" try: text = Element("text-input").value doc = nlp(text) frases = [s.text.strip() for s in doc.sents if s.text.strip()] if not frases: status_el.write("Nenhuma frase encontrada.") return df = calcular_icoer(frases) table_el.write(df.to_html(index=False), append=False) dfp = df[df['narraton']!=0] if not dfp.empty: fig, ax = plt.subplots(figsize=(10,6)) labels = [f"{i}: {f[:30]}…" for i,f in zip(dfp['narraton'], dfp['frase'])] ax.bar(labels, dfp['icoer']) ax.set_ylim(0,1.1) plt.xticks(rotation=45, ha="right") plt.tight_layout() buf = io.BytesIO(); fig.savefig(buf, format="png"); buf.seek(0) img = base64.b64encode(buf.read()).decode("utf-8") plot_el.write(f'', append=False) plt.close(fig) status_el.write("Concluído.") except Exception as e: status_el.write(f"Erro: {e}") finally: loader_el.element.style.display = "none" # dispara o carregamento inicial do spaCy await setup_spacy()
