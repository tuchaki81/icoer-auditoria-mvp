Ferramenta de Análise ICOER

Cole seu texto abaixo e clique em “Analisar Texto”.

Desenvolvido com Python, PyScript e SpaCy.
packages = ["pandas","numpy","matplotlib","micropip"] files = [ "https://github.com/explosion/spacy-models/releases/download/pt_core_news_sm-3.7.0/pt_core_news_sm-3.7.0-py3-none-any.whl" ] import asyncio, micropip from pyscript import Element import pandas as pd, numpy as np import matplotlib.pyplot as plt from collections import Counter import io, base64 # variáveis globais nlp = None BASE_POS = ["NOUN","VERB","ADJ","ADV","ADP","PROPN"] # 1) carrega o spaCy e o modelo de português async def init(): status = Element("output-status") loader = Element("loader") status.write("Carregando modelo de linguagem…") loader.element.style.display = "block" await micropip.install("spacy") # instala o core do spaCy # nossa wheel já foi baixada via , basta instalar await micropip.install("pt_core_news_sm-3.7.0-py3-none-any.whl") import spacy global nlp nlp = spacy.load("pt_core_news_sm") status.write("Modelo carregado! Pronto para análise.") loader.element.style.display = "none" # dispara o init em background sem usar await no topo asyncio.create_task(init()) # 2) função que calcula o ICOER def calcular_icoer(frases): registros = [] for i, sent in enumerate(nlp.pipe(frases), start=1): if len(sent) < 3: continue counts = Counter([t.pos_ for t in sent]) total = sum(counts.values()) or 1 probs = np.array([counts[p]/total for p in BASE_POS if counts[p]>0]) ent = -np.sum(probs * np.log2(probs)) if probs.size>0 else 0 ent_max = np.log2(len(BASE_POS)) icoer = round(max(0, 1 - (ent/ent_max if ent_max>0 else 0)), 4) registros.append({ "narraton": i, "frase": sent.text, "n_palavras": len(sent), "tags": dict(counts), "icoer": icoer }) if not registros: return pd.DataFrame([{"narraton":0,"frase":"NENHUMA FRASE VÁLIDA","n_palavras":0,"tags":{},"icoer":0}]) df = pd.DataFrame(registros) resumo = pd.DataFrame([{ "narraton": 0, "frase": "MÉDIA GERAL", "n_palavras": round(df["n_palavras"].mean(),2), "tags": {p: round(df["tags"].apply(lambda d: d.get(p,0)).mean(),2) for p in BASE_POS}, "icoer": round(df["icoer"].mean(),4) }]) return pd.concat([resumo, df], ignore_index=True) # 3) função chamada no clique — deve ser SÍNCRONA def run_analysis(*args, **kwargs): status = Element("output-status") table = Element("output-table") plot = Element("output-plot") loader = Element("loader") if nlp is None: status.write("Ainda carregando o modelo, aguarde…") return text = Element("text-input").value frases = [s.text.strip() for s in nlp(text).sents if s.text.strip()] if not frases: status.write("Nenhuma frase encontrada.") return status.write("Analisando…") loader.element.style.display = "block" df = calcular_icoer(frases) table.write(df.to_html(index=False), append=False) dfp = df[df["narraton"]!=0] if not dfp.empty: fig, ax = plt.subplots(figsize=(8,4)) ax.bar(dfp["narraton"].astype(str), dfp["icoer"]) ax.set_ylim(0,1) ax.set_xlabel("Narraton") ax.set_ylabel("ICOER") plt.tight_layout() buf = io.BytesIO() fig.savefig(buf, format="png"); buf.seek(0) img = base64.b64encode(buf.read()).decode("utf-8") plot.write(f'', append=False) plt.close(fig) status.write("Pronto!") loader.element.style.display = "none"
