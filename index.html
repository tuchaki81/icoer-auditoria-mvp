Ferramenta de Análise ICOER

Bem-vindo à ferramenta de Análise de Coerência Informacional (ICOER). Esta página permite que você analise a coerência de textos em português com base na distribuição de classes gramaticais (POS-Tagging).

O que é o ICOER?

O Índice de Coerência Informacional (ICOER) é uma métrica que mede a coerência de uma frase. Valores próximos de 1 indicam alta coerência e foco temático, enquanto valores próximos de 0 sugerem maior diversidade gramatical ou dispersão. É uma ferramenta útil para auditoria de conteúdo, análise de textos gerados por IA e estudos linguísticos.

Como Usar

Cole seu texto no campo abaixo. A ferramenta irá separar as frases, calcular o ICOER para cada uma e exibir os resultados em uma tabela e um gráfico.

# Lista de pacotes que o PyScript precisa baixar. packages = [ "pandas", "numpy", "matplotlib", "spacy" ] # Lista de arquivos (wheels) que o PyScript deve baixar antes de executar files = [ "https://github.com/explosion/spacy-models/releases/download/pt_core_news_sm-3.7.0/pt_core_news_sm-3.7.0-py3-none-any.whl" ] import pandas as pd import numpy as np import spacy from collections import Counter import matplotlib.pyplot as plt import io import base64 from pyscript import Element status_element = Element("output-status") table_element = Element("output-table") plot_element = Element("output-plot") loader_element = Element("loader") nlp = None BASE_POS = ['NOUN', 'VERB', 'ADJ', 'ADV', 'ADP', 'PROPN'] async def setup_spacy(): global nlp if nlp is None: status_element.write("Carregando modelo de linguagem (pode levar um minuto)...") loader_element.element.style.display = "block" try: # Instala o modelo a partir do wheel baixado await spacy.cli.package.install("pt_core_news_sm-3.7.0-py3-none-any.whl") nlp = spacy.load("pt_core_news_sm") status_element.write("Modelo carregado com sucesso! Pronto para análise.") except Exception as e: status_element.write(f"Erro ao carregar o modelo: {e}") finally: loader_element.element.style.display = "none" def calcular_icoer(frases): docs = list(nlp.pipe(frases)) registros = [] for i, doc in enumerate(docs, start=1): if not doc.text.strip() or len(doc) < 3 or not any(t.is_alpha or t.is_digit for t in doc): continue counts = Counter(t.pos_ for t in doc) total = sum(counts.values()) vetor = [counts.get(pos, 0)/total if total > 0 else 0 for pos in BASE_POS] probs = np.array([v for v in vetor if v > 0]) ent = -np.sum(probs * np.log2(probs)) if probs.size else 0 ent_max = np.log2(len(BASE_POS)) if len(BASE_POS) > 1 else 1 icoer = round(max(0, 1 - (ent/ent_max if ent_max > 0 else 0)), 4) registros.append({ 'narraton': i, 'frase': doc.text, 'n_palavras': len(doc), 'tags': dict(counts), 'icoer': icoer }) df = pd.DataFrame(registros) if df.empty: return pd.DataFrame([{ 'narraton': 0, 'frase': 'NENHUMA FRASE VÁLIDA', 'n_palavras': 0, 'tags': {}, 'icoer': 0 }]) # Linha de resumo tags_summary = { pos: round(np.mean([r['tags'].get(pos, 0) for r in registros]), 2) for pos in BASE_POS } resumo = pd.DataFrame([{ 'narraton': 0, 'frase': 'MÉDIA GERAL', 'n_palavras': round(df['n_palavras'].mean(), 2), 'tags': tags_summary, 'icoer': round(df['icoer'].mean(), 4) }]) return pd.concat([resumo, df], ignore_index=True)[['narraton','frase','n_palavras','tags','icoer']] async def run_analysis(*args, **kwargs): if nlp is None: await setup_spacy() if nlp is None: return table_element.clear() plot_element.clear() status_element.write("Analisando...") loader_element.element.style.display = "block" try: raw_text = Element("text-input").value doc = nlp(raw_text) frases = [sent.text.strip() for sent in doc.sents if sent.text.strip()] if not frases: status_element.write("Nenhuma frase encontrada para análise.") return df_result = calcular_icoer(frases) table_element.write(df_result.to_html(index=False), append=False) df_plot = df_result[df_result['narraton'] != 0] if not df_plot.empty: fig, ax = plt.subplots(figsize=(10, 6)) labels = [f"{i}: {s[:30]}..." for i, s in zip(df_plot['narraton'], df_plot['frase'])] ax.bar(labels, df_plot['icoer']) ax.set_xlabel('Narraton (Frase)') ax.set_ylabel('ICOER (0 a 1)') ax.set_title('Índice de Coerência Informacional por Frase') plt.xticks(rotation=45, ha="right") ax.set_ylim(0, 1.1) plt.tight_layout() buf = io.BytesIO() fig.savefig(buf, format="png") buf.seek(0) img_str = base64.b64encode(buf.read()).decode("utf-8") plot_element.write(f'', append=False) plt.close(fig) status_element.write("Análise concluída.") except Exception as e: status_element.write(f"Ocorreu um erro durante a análise: {e}") finally: loader_element.element.style.display = "none" await setup_spacy()
